set.seed(2020)

svals<- svd(X)$d

X <- cbind(rep(1,5), X)
beta <- c(0.1, 100,1)
mu <- 1 / (1+exp(-X %*% beta))
D <- diag(x=as.vector(mu), nrow=5, ncol=5)
